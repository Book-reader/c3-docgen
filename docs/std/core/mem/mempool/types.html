FixedBlockPool

Fixed blocks pool pre-allocating blocks backed by an Allocator which are then reserved for the user,blocks deallocated by the user are later re-used by future blocks allocations`grow_capacity` can be changed in order to affect how many blocks will be allocated by next pool allocation,it has to be greater than 0`allocated` number of allocated blocks`used` number of used blocks by the user
struct FixedBlockPool
{
	Allocator allocator;
	FixedBlockPoolNode head;
	FixedBlockPoolNode *tail;
	void *next_free;
	void *freelist;
	usz block_size;
	usz grow_capacity;
	usz allocated;
	usz page_size;
	usz alignment;
	usz used;
	bool initialized;
}

FixedBlockPoolEntry

struct FixedBlockPoolEntry
{
	void *previous;
}

FixedBlockPoolNode

struct FixedBlockPoolNode
{
	void* buffer;
	FixedBlockPoolNode *next;
	usz capacity;
}
